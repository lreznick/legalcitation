We assume that you have root access for all of the installation process

1. Get Apache 2
	sudo apt-get install apache2
	
2. Get Pip
	sudo apt-get python-pip
	
3. Get web.py
	sudo pip install web.py

4. Get passlib
	sudo apt-get install python-passlib

5. Get MySQL
	sudo apt-get install python-mysql.connector
	sudo apt-get install python-mysqldb
	
4. Get mod_wsgi
	sudo apt-get install libapache2-mod-wsgi
	
	4.0 Check to see its working
		sudo service apache2 restart
		put ip address in web browser to see if it works

5. enable mod_rewrite
	a2enmod rewrite
	
6. Configuration of apache
	cd /etc/apache2/sites-available

	6.1 Test to see if there are errors
		apachectl configtest
	6.2 Monitor For errors when messing with config files (optional)
		tail -f /var/log/apache2/error.log
		and 

	
	7.2 Creating a .htaccess file
		sudo vim .htaccess
		Added:
			<Files index.py>
				SetHandler wsgi-script
				Options +ExecCGI +FollowSymLinks
			</Files>
			RewriteEngine On
			RewriteBase /
			RewriteCond %{REQUEST_URI} !^favicon.ico$
			RewriteCond %{REQUEST_URI} !^(/.*)+index.py/
			RewriteCond %{REQUEST_URI} !^(/.*)+\.(css|png|js|jpg|tff|oft|gif)$
			RewriteRule ^(.*)$ index.py/$1 [PT]

		chown -R 755 /var/www/html 
	8.2 Connecting in Filezilla
		Open Site manager
			host: some ip like 54.213.173.233
			Port: 22

			Login Type: Normal

	sudo mkdir html

	sudo mkdir test.com

	sudo vim index.py (or whatever your main is)
